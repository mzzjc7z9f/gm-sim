  return {shoot3:s.shoot3/sum,shoot2:s.shoot2/sum,drive:s.drive/sum,pass:s.pass/sum,post:s.post/sum};
}

// ── スタッツ生成（引き継ぎ書仕様） ──────────────────────────
function calcPlayerStats(player, team, minutes) {
  const a=player.abilities, sty=player.style, sys=team.system, pos=player.position;
  const poss = sys.pace*(minutes/48);
  const usage = clamp(0.18+(a.scoring-70)*0.004, 0.15, 0.35);
  const shots = poss*usage;
  const actualShots = shots*(1-sty.pass);
  const threeRate = clamp(sty.shoot3+sys.threeRate*0.15, 0.10, 0.68);
  const tpa=actualShots*threeRate, twpa=actualShots*(1-threeRate);
  const tp_pct  = clamp(0.32+(a.scoring-70)*0.003+(a.iq-70)*0.002, 0.25, 0.45);
  const two_pct = clamp(0.48+(a.physical-70)*0.003, 0.40, 0.65);
  const tpm=tpa*tp_pct, twpm=twpa*two_pct;
  const fga=tpa+twpa, fgm=tpm+twpm, fg_pct=fga>0?fgm/fga:0;
  const foulRate=clamp(0.10+(a.scoring-70)*0.0025,0.05,0.22);
  const ft_pct=clamp(0.68+(a.iq-70)*0.003,0.50,0.95);
  const fta=actualShots*foulRate*2.0, ftm=fta*ft_pct;
  const pts=tpm*3+twpm*2+ftm;
  const playmakingCoeff=clamp(0.5+(a.playmaking-50)/100*1.5,0.5,2.0);
  const ast=shots*sty.pass*0.7*playmakingCoeff;
  const missedShots=(tpa-tpm)+(twpa-twpm);
  const rebBaseRate={PG:0.08,SG:0.10,SF:0.14,PF:0.22,C:0.28}[pos]||0.12;
  const physCoeff=clamp(0.5+(a.physical-50)/100,0.5,1.5);
  const teamMissed=sys.pace*0.87*0.53*(minutes/48);
  const reb=(missedShots+teamMissed)*rebBaseRate*physCoeff;
  const stlBase={PG:1.6,SG:1.3,SF:1.2,PF:0.9,C:0.7}[pos]||1.0;
  const stl=stlBase*(minutes/36)*clamp(0.5+(a.defense-50)/100+(a.iq-50)/200,0.4,1.8);
  const blkBase={PG:0.2,SG:0.3,SF:0.6,PF:1.2,C:2.0}[pos]||0.5;
  const blk=blkBase*(minutes/36)*clamp(0.4+(a.defense-50)/100+(a.physical-50)/150,0.3,2.0);
  const tov=clamp(actualShots*0.12*(1.4-a.playmaking/100)*(1.2-a.iq/200),0.3,4.5)*(minutes/36);
  const pm=clamp((calcOVR(a)-70)*0.08+rand(-2,2),-12,15);
  return {min:Math.round(minutes),pts:Math.max(0,pts),reb:Math.max(0,reb),ast:Math.max(0,ast),
    tpa,tpm,tp_pct,twpa,twpm,two_pct,fga,fgm,fg_pct,usage,
    fta:Math.max(0,fta),ftm:Math.max(0,ftm),ft_pct,
    stl:Math.max(0,stl),blk:Math.max(0,blk),tov:Math.max(0,tov),pm};
}

function makePlayer(ovrTarget, age, pos) {
  const position = pos||POSITIONS[randInt(0,4)];
  const target   = clamp(ovrTarget||randInt(58,82),40,99);
  const profile  = {PG:[0.85,0.60,1.30,0.65,1.05],SG:[1.10,0.75,0.95,0.80,0.85],SF:[1.00,0.90,0.80,1.00,0.80],PF:[0.90,1.00,0.60,1.20,0.70],C:[0.70,1.10,0.45,1.30,0.70]}[position];
  const keys=['scoring','defense','playmaking','physical','iq'];
  const weights=[0.3,0.2,0.2,0.2,0.1];
  const raw={};
  keys.forEach((k,i)=>{raw[k]=clamp(Math.round(target*profile[i]+rand(-6,6)),40,99);});
  for(let iter=0;iter<15;iter++){const diff=target-calcOVR(raw);if(Math.abs(diff)<=1)break;keys.forEach((k,i)=>{raw[k]=clamp(raw[k]+Math.round(diff*weights[i]/weights[0]*0.5),40,99);});}
  const stRaw={shoot3:rand(0.05,pos==='PG'||pos==='SG'?0.50:0.35),shoot2:rand(0.10,0.40),drive:rand(0.05,pos==='PG'?0.40:0.30),pass:rand(0.05,pos==='PG'?0.35:0.20),post:rand(0.02,pos==='C'||pos==='PF'?0.35:0.15)};
  const style=normalizeStyle(stRaw);
  const potMax=age<=20?1.4:age<=23?1.3:age<=26?1.15:1.0;
  const potential=clamp(rand(0.75,potMax),0.75,1.4);
  return {id:uid(),name:rName(),age:age||randInt(19,34),position,abilities:raw,style,potential,peakAge:randInt(26,31),contract:makeContract(age||24,calcOVR(raw)),stats:null};
}

function makeContract(age,ovr) {
  const CAP_V=CAP_NEW||140_588_000, MIN_S=MIN_SALARY||1_119_563;
  if(ovr<70||age>=35) return {salary:Math.round(clamp(CAP_V*0.013,MIN_S,MIN_S*3)),years:1,type:'minimum'};
  const value=ovr*1.5;
  const maxPct=ovr>=90?0.35:ovr>=80?0.30:ovr>=75?0.25:0.20;
  const salary=Math.round(clamp(50_000_000*value/100,MIN_S*3,CAP_V*maxPct));
  let years,type;
  if(age<=22)            {years=4;type='rookie';}
  else if(age<=27&&ovr>=80){years=randInt(4,5);type='extension';}
  else if(age<=30)       {years=randInt(2,4);type='extension';}
  else if(age<=34)       {years=randInt(1,3);type='extension';}
  else                   {years=1;type='minimum';}
  return {salary,years,type};
}

function makeTeam(def, idx) {
  const strength = def.strength || 80;

  const CAP_V    = CAP_NEW || 140_588_000;
  const posDist  = ['PG','PG','PG','SG','SG','SG','SF','SF','SF','PF','PF','PF','C','C','C'];

  const ovrDist = [
    starOVR,

    clamp(strength + randInt(-4,2), 70, 89),
    clamp(strength + randInt(-5,1), 68, 87),

    clamp(strength + randInt(-11,-5),61,79),

    clamp(strength + randInt(-13,-7),58,76),
    clamp(strength + randInt(-14,-8),56,74),
    clamp(strength + randInt(-15,-9),54,72),
    clamp(strength + randInt(-16,-10),53,70),
    clamp(strength + randInt(-17,-11),52,68),
    clamp(strength + randInt(-18,-12),50,66),
  ];

  const players = posDist.map((pos, i) => {
    const tgt    = ovrDist[i] || clamp(strength - 15, 50, 70);
    const ageBase = strength >= 88 ? randInt(24,32) : strength >= 82 ? randInt(22,30) : randInt(19,27);
    return makePlayer(tgt, ageBase, pos);
  });
  players.sort((a,b) => calcOVR(b.abilities)-calcOVR(a.abilities));

  const SALARY_TIERS = [
    [0.27,0.35],[0.16,0.24],[0.10,0.17],[0.07,0.12],[0.05,0.09],

  ];
  players.forEach((p, i) => {
    const tier = SALARY_TIERS[Math.min(i, SALARY_TIERS.length-1)];

    ...def, idx, players, system,
    wins:0, losses:0,
    revenue: randInt(180,340)*1_000_000,
    expenses: randInt(20,55)*1_000_000,
    get payroll() { return this.players.reduce((s,p)=>s+p.contract.salary,0); }
  };
}

function loadGame() {
  try {
    const raw = localStorage.getItem('gmSim_save');
    if(!raw) return false;
    const data = JSON.parse(raw);
    if(!data.G) return false;
    // Gを復元
    Object.assign(G, data.G);
    // payrollはgetterとして再定義
    G.teams.forEach(t => {

          get() { return this.players.reduce((s,p)=>s+p.contract.salary,0); },
          configurable: true, enumerable: false
        });
      }
    });
    return true;
  } catch(e) {
    return false;
  }
}

const INJURY_TYPES = [
  {name:'足首捻挫',   weeks:[2,4],  prob:0.06},
  {name:'ハムストリング', weeks:[3,6], prob:0.04},
  {name:'膝の打撲',   weeks:[1,3],  prob:0.05},
  {name:'肩脱臼',    weeks:[4,8],  prob:0.02},
  {name:'疲労骨折',   weeks:[8,14], prob:0.01},
  {name:'ACL断裂',   weeks:[30,52],prob:0.005},
];

function rollInjury(player) {
  // 高齢・低身体能力ほど怪我しやすい
  const basePct = 0.008 + Math.max(0, player.age-28)*0.003 + Math.max(0,70-player.abilities.physical)*0.002;
  if(Math.random() > basePct) return null;
  const pool = INJURY_TYPES.filter(t => Math.random() < t.prob*5);
  if(!pool.length) return null;
  const injury = pool[Math.floor(Math.random()*pool.length)];
  const weeks  = injury.weeks[0] + Math.floor(Math.random()*(injury.weeks[1]-injury.weeks[0]+1));
  return {type: injury.name, weeksOut: weeks, weeksRemaining: weeks};
}

function processInjuries(team) {
  let newsItems = [];
  team.players.forEach(p => {
    // 既存怪我の回復
    if(p.injury) {
      p.injury.weeksRemaining = Math.max(0, p.injury.weeksRemaining - 1);
      if(p.injury.weeksRemaining === 0) {
        newsItems.push({type:'good', title:p.name+' 復帰', body:p.name+' が'+p.injury.type+'から回復しました。'});
        delete p.injury;
      }
      return;
    }
    // 新規怪我
    const inj = rollInjury(p);
    if(inj) {
      p.injury = inj;
      const severity = inj.weeksOut >= 20 ? 'major' : inj.weeksOut >= 8 ? 'warn' : 'info';
      newsItems.push({type: severity, title:p.name+' 負傷', body:p.name+' が'+inj.type+'で'+inj.weeksOut+'週間の離脱。'});
    }
  });
  return newsItems;
}

function getInjuredPlayers(team) {
  return team.players.filter(p => p.injury && p.injury.weeksRemaining > 0);
}

  const ovr = teamAvgOVR(t);
  const pay = t.payroll;
  const cap = CAP_NEW||140_588_000;

  let score = 50; // ベース

  // 勝率ボーナス
  score += Math.round((winPct - 0.5) * 60);
  // カンファレンス順位ボーナス
  score += Math.max(0, 16 - rank * 1.5);
  // チームOVRボーナス
  score += Math.round((ovr - 78) * 1.5);

  if(pay < (TAX_LINE||170_814_000)) score += 5;
  // 2nd Apron超過はペナルティ
  if(pay > (APRON_2||188_931_000)) score -= 10;

  return Math.round(clamp(score, 0, 100));
}

// ── 成長・衰退（引き継ぎ書仕様） ───────────────────────────
function applyGrowth() {
  G.teams.forEach(t=>{
    const expired=[];
    t.players.forEach(p=>{
      if(p.stats&&p.stats.pts>0){
        if(!p.statsHistory) p.statsHistory=[];
        p.statsHistory.push({season:G.season,team:t.abbr,age:p.age,ovr:calcOVR(p.abilities),...p.stats});
        if(p.statsHistory.length>15) p.statsHistory.shift();
      }
      p.age++; p.contract.years=Math.max(0,p.contract.years-1);
      const a=p.abilities;
      let delta=0;
      if(p.age<=24){
        const ageCoeff=clamp(1.0-(p.age-18)*0.12,0.28,1.0);
        delta=Math.round(p.potential*ageCoeff*4+rand(-0.5,1.5));
      } else if(p.age<=29){
        delta=Math.round(rand(-0.8,0.8));
      } else {
        delta=Math.round(-(p.age-29)*0.7+rand(-0.5,0.5));
      }
      ['scoring','defense','playmaking','physical','iq'].forEach(k=>{
        a[k]=clamp(a[k]+delta+randInt(-1,1),40,99);
      });
      if(p.contract.years===0) expired.push(p);
    });
    t.players=t.players.filter(p=>p.contract.years>0);

    // 満了選手の処理
    expired.forEach(p=>{
      const ovr=calcOVR(p.abilities);
      // 自チームは自動再契約しない（プレイヤーが決める）
      if(t.idx===G.myIdx){
        G.freeAgents.unshift(p); // FAリスト先頭に追加（目立つように）
      } else {
        // CPU: OVR75以上・35歳未満は65%で自動再契約
        if(ovr>=75&&p.age<35&&Math.random()<0.65){
          p.contract=makeContract(p.age,ovr); t.players.push(p);
        } else {
          G.freeAgents.push(p);
        }
      }
    });

    // ── ロスター補充（CPU限定）── 最低8人を確保
    if(t.idx!==G.myIdx && t.players.length<8){
      const needed=8-t.players.length;
      const pool=[...G.freeAgents].sort(()=>Math.random()-0.5);
      for(let i=0;i<Math.min(needed,pool.length);i++){
        const fa=pool[i];
        fa.contract=makeContract(fa.age,calcOVR(fa.abilities));
        t.players.push(fa);
        G.freeAgents=G.freeAgents.filter(f=>f.id!==fa.id);
      }
      // それでも足りなければ新たに生成
      while(t.players.length<8){
        const np=makePlayer(randInt(58,72),randInt(22,32));
        np.contract=makeContract(np.age,calcOVR(np.abilities));
        t.players.push(np);
      }
    }

    // 自チームが2人以下ならFAから自動補充（最低限のUI維持）
    if(t.idx===G.myIdx && t.players.length<2){
      addLog('!! ロスターが'+t.players.length+'人。FAから緊急補充します','bad');
      const pool=[...G.freeAgents].sort((a,b)=>calcOVR(b.abilities)-calcOVR(a.abilities));
      for(let i=0;i<Math.min(5,pool.length)&&t.players.length<5;i++){
        const fa=pool[i];
        fa.contract=makeContract(fa.age,calcOVR(fa.abilities));
        t.players.push(fa);
        G.freeAgents=G.freeAgents.filter(f=>f.id!==fa.id);
      }
    }
  });
}

function aiRunDraft() {
  // ドラフト順に全チームが1人ずつ指名
  const order=G.draftOrder||G.teams.map(t=>t.idx);
  order.forEach(idx=>{
    if(idx===G.myIdx) return; // プレイヤーは手動
    const t=G.teams[idx];
    if(!t||t.players.length>=15) return;
    const rebuild=teamAvgOVR(t)<68;
    const avail=G.draftProspects.filter(p=>!p.drafted);
    if(!avail.length) return;
    avail.sort((a,b)=>{
      const sa=calcOVR(a.abilities)+(rebuild?a.potential*12:0);
      const sb=calcOVR(b.abilities)+(rebuild?b.potential*12:0);
      return sb-sa;
    });
    const pick=avail[0];
    if(!pick) return;
    pick.drafted=true;
    pick.contract=makeContract(pick.age,calcOVR(pick.abilities));
    t.players.push(pick);
  });
}

  tiers.forEach(({n, r, tier}) => {
    for(let i = 0; i < n; i++) {
      const age = pick <= 10 ? randInt(18,21) : pick <= 30 ? randInt(18,22) : randInt(19,23);
      const p = makePlayer(randInt(...r), age);
      p.tier = tier; p.draftPos = pick++; p.drafted = false;
      p.round = p.draftPos <= 30 ? 1 : 2;
      // バスト（5%）: 期待外れ
      if(p.draftPos <= 10 && Math.random() < 0.08) {
        p.tier = 'Bust';
        ['scoring','defense','playmaking','physical','iq'].forEach(k => { p.abilities[k] = clamp(p.abilities[k] - randInt(8,15), 40, 70); });
      }
      // スリーパー（2%）: 低評価の原石
      if(p.draftPos >= 40 && Math.random() < 0.04) {
        p.tier = 'Sleeper';
        ['scoring','defense','playmaking','physical','iq'].forEach(k => { p.abilities[k] = clamp(p.abilities[k] + randInt(8,16), 40, 99); });
      }
      G.draftProspects.push(p);
    }
  });
}

function teamAvgOVR(t) {
  if(!t.players.length) return 60;
  // 上位8人の平均（NBAで「チーム戦力」はローテーション選手で判断）
  const sorted = [...t.players].sort((a,b)=>calcOVR(b.abilities)-calcOVR(a.abilities));
  const rotation = sorted.slice(0, Math.min(8, sorted.length));
  return rotation.reduce((s,p)=>s+calcOVR(p.abilities),0)/rotation.length;
}

function aiRunFreeAgency() {
  G.teams.forEach((t,i)=>{
    if(i===G.myIdx) return;
    const space=CAP-t.payroll, isContend=teamAvgOVR(t)>=73;
    const sorted=[...G.freeAgents].sort((a,b)=>{const sa=calcOVR(a.abilities)+(isContend?0:a.potential*8),sb=calcOVR(b.abilities)+(isContend?0:b.potential*8);return sb-sa;});
    for(const fa of sorted){
      if(t.players.length>=15) break;
      if(fa.contract.salary>Math.max(space,3_000_000)) continue;
      if(Math.random()<0.55){fa.contract=makeContract(fa.age,calcOVR(fa.abilities));t.players.push(fa);G.freeAgents=G.freeAgents.filter(f=>f.id!==fa.id);}
    }
  });
}

function mkTr(cells) {
  const tr = document.createElement('tr');
  cells.forEach(c => {
    const td = document.createElement('td');
    if(typeof c === 'string') td.innerHTML = c;
    else if(c instanceof HTMLElement) td.appendChild(c);

    // Anthropic API呼び出し
    fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': '',
        'anthropic-version': '2023-06-01',
        'anthropic-dangerous-direct-browser-access': 'true',
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 300,
        system: 'あなたはNBAの専門用語を日本語でわかりやすく解説するアシスタントです。GMシミュレーターゲームのプレイヤーに向けて、「'+term+'」を3〜5文で簡潔に説明してください。ゲーム内での影響（キャップへの影響、使える例外など）も含めてください。説明のみを返してください。余分な前置きは不要です。',

// ラグジュアリータックス累進計算
function calcLuxTax(pay, repeater) {
  if(pay <= (TAX_LINE||170_814_000)) return 0;
  const over = pay - (TAX_LINE||170_814_000);
  const brackets = [5e6, 5e6, 5e6, 5e6, Infinity];
  const rates    = repeater ? [2.50, 2.75, 3.50, 4.25, 4.75] : [1.50, 1.75, 2.50, 3.25, 3.75];
  let tax = 0, rem = over;
  brackets.forEach((b, i) => {
    if(rem <= 0) return;
    const chunk = Math.min(rem, b);
    tax += chunk * rates[i];
    rem -= chunk;
  });
  return Math.round(tax);
}

// サラリーマッチング (NBA 125%+10万ルール)
function tradeSalCheck(sendSal, recvSal, useTpe) {
  const me = G.teams[G.myIdx];
  if(!me.tpe) me.tpe = null;
  const tpeSal = (useTpe && me.tpe) ? me.tpe.salary : 0;
  const effective = sendSal + tpeSal;
  const limit = effective * 1.25 + 100_000;
  if(recvSal <= limit) {
    const tag = tpeSal>0 ? ' (TPE '+fmtM(tpeSal)+')' : '';
    return {ok:true, msg:'Salary OK'+tag};
  }
  const gap = recvSal - limit;
  return {ok:false, msg:'差額 '+fmtM(gap)+' 超過'};
}

// 指名権価値 (NBA現実準拠: 1位=$75M, 5位=$45M, 10位=$24M, 30位=$5M)
// ══════════════════════════════════════════════════════════════
// GM BRAIN — このゲーム専用AIエンジン
// チームの状況を分析して意思決定を行う
// ══════════════════════════════════════════════════════════════
const GMBrain = {

  // チームフェーズ判定
  teamPhase(team) {
    const ovr = teamAvgOVR(team);
    const wins = team.wins || 0;
    const losses = team.losses || 1;
    const winPct = wins / (wins + losses);
    const corePlayers = team.players.filter(p => calcOVR(p.abilities) >= 78);
    const youngStars  = team.players.filter(p => calcOVR(p.abilities) >= 75 && p.age <= 25);

    if(ovr >= 78 && winPct >= 0.55) return 'contend';      // 優勝狙い
    if(ovr >= 74 && winPct >= 0.45) return 'compete';      // プレーオフ争い
    if(youngStars.length >= 2)       return 'build';        // 若手育成中
    if(ovr < 68 || winPct < 0.35)   return 'rebuild';      // 再建中
    return 'neutral';
  },

  // 選手の「このチームにとっての価値」
  playerNeed(team, player) {
    const phase = this.teamPhase(team);
    const ovr   = calcOVR(player.abilities);
    const age   = player.age;
    const pos   = player.position;

    // ポジション補完（足りないポジションは高評価）
    const posCount = team.players.filter(p => p.position === pos).length;
    const posMult  = posCount <= 1 ? 1.15 : posCount >= 4 ? 0.88 : 1.0;

    // フェーズによる年齢・タイプ優先度
    let phaseMult = 1.0;
    if(phase === 'contend') {
      phaseMult = age <= 30 && ovr >= 76 ? 1.20 : age >= 32 ? 0.85 : 1.0;
    } else if(phase === 'rebuild') {
      phaseMult = age <= 23 ? 1.30 : age >= 30 ? 0.70 : 1.0;
    } else if(phase === 'build') {
      phaseMult = age <= 25 ? 1.25 : age >= 32 ? 0.80 : 1.0;
    }

    return posMult * phaseMult;
  },

  // 選手のトレード市場価値（GMBrain版）
  tradeValue(player, forTeam) {
    const ovr  = calcOVR(player.abilities);
    const age  = player.age;
    const pot  = player.potential;
    const sal  = player.contract.salary;
    const yrs  = player.contract.years;

    // OVRベース価値テーブル（現実NBA準拠）
    const BASE = [
      [95,120],[92,95],[89,75],[86,58],[83,42],[80,30],
      [77,20],[74,13],[71,8],[68,5],[65,3],[0,1.5]
    ];
    let base = 1_500_000;
    for(const [threshold, val] of BASE) {
      if(ovr >= threshold) { base = val * 1_000_000; break; }
    }

    // 年齢補正
    let ageMult;
    if(age <= 21)      ageMult = 1.35;
    else if(age <= 23) ageMult = 1.22;
    else if(age <= 25) ageMult = 1.12;
    else if(age <= 28) ageMult = 1.00;
    else if(age <= 30) ageMult = 0.92;
    else if(age <= 32) ageMult = 0.80;
    else if(age <= 34) ageMult = 0.65;
    else               ageMult = 0.45;

    // ポテンシャル補正（26歳以下限定）
    const potMult = age <= 26 ? clamp(0.85 + (pot - 0.75) * 0.5, 0.85, 1.40) : 1.0;

    // 契約補正: 残年数が多いほど価値UP（若手）or DOWN（ベテラン）
    let contractMult = 1.0;
    const capPct = sal / (CAP_NEW || 140_588_000);
    if(age <= 26 && yrs >= 3) contractMult = 1.10;  // 若手の長期契約は価値高
    if(age >= 32 && capPct > 0.15) contractMult = 0.82; // 老ベテランの高給は価値低

    let value = base * ageMult * potMult * contractMult;

    // 需要補正
    if(forTeam) value *= this.playerNeed(forTeam, player);

    return Math.round(value);
  },

  // 指名権価値（年度・順位による）
  pickValue(round, fromTeamIdx, yearOffset) {
    const offset = yearOffset || 0;
    const ranked = [...G.teams].sort((a,b) => a.wins - b.wins);
    const pos    = clamp(ranked.findIndex(t => t.idx === fromTeamIdx) + 1, 1, 30);
    // 将来の指名権は不確実性discount
    const futureDiscount = Math.pow(0.88, offset);
    if(round === '1st') {
      // 1位=$75M、指数減衰、最低$4M
      const val = Math.max(4_000_000, 75_000_000 * Math.pow(0.87, pos - 1));
      return Math.round(val * futureDiscount);
    }
    return Math.round(Math.max(800_000, 3_500_000 * Math.pow(0.93, pos - 1)) * futureDiscount);
  },

  // AIトレード受諾判定（相手GMとして）
  shouldAcceptTrade(myTeam, theirTeam, iReceive, iGive) {
    // iReceive/iGive: {players:[], picks:[]}
    const phase = this.teamPhase(myTeam);

    const recvVal = iReceive.players.reduce((s,p) => s + this.tradeValue(p, myTeam), 0)
                  + iReceive.picks.reduce((s,k) => s + this.pickValue(k.round, k.fromTeamIdx), 0);
    const giveVal = iGive.players.reduce((s,p) => s + this.tradeValue(p, theirTeam), 0)
                  + iGive.picks.reduce((s,k) => s + this.pickValue(k.round, k.fromTeamIdx), 0);

    if(recvVal <= 0 || giveVal <= 0) return {ok:false, reason:'資産がありません'};

    const ratio = recvVal / Math.max(giveVal, 1);

    // 価値比率チェック（NBA現実: 0.75倍以上なら検討）
    if(ratio < 0.72) return {ok:false, reason:'価値が釣り合いません（'+Math.round(ratio*100)+'%）'};
    if(ratio > 2.80) return {ok:false, reason:'こちらが払いすぎです'};

    // フェーズ整合性チェック
    const recvPlayers = iReceive.players;
    const givePlayers = iGive.players;

    if(phase === 'rebuild') {
      const giveYoung = givePlayers.filter(p => p.age <= 24 && calcOVR(p.abilities) >= 70).length;
      const givePickCount = iGive.picks.filter(k => k.round === '1st').length;
      if(giveYoung + givePickCount === 0 && givePlayers.length > 0) {
        return {ok:false, reason:'再建中: 若手か1巡目指名権が必要'};
      }
    }

    if(phase === 'contend') {
      const recvStar = recvPlayers.some(p => calcOVR(p.abilities) >= 78);
      const giveStar = givePlayers.some(p => calcOVR(p.abilities) >= 78);
      if(giveStar && !recvStar && iReceive.picks.length === 0) {
        return {ok:false, reason:'優勝狙い: スターの対価が不十分'};
      }
    }

    // サラリーバランス確認
    const salIn  = iReceive.players.reduce((s,p) => s + p.contract.salary, 0);
    const salOut = iGive.players.reduce((s,p) => s + p.contract.salary, 0);
    const newPay = myTeam.payroll - salOut + salIn;
    if(newPay > (APRON_2 || 188_931_000)) {
      return {ok:false, reason:'2nd Apron超過になります'};
    }

    // ランダム要素（リアルなGMの気まぐれ）
    if(ratio >= 0.88 && Math.random() < 0.15) return {ok:false, reason:'現時点では合意に至りませんでした'};

    return {ok:true, reason:''};
  },

  // AI発案: このチームが行うべきトレード案を生成
  proposeTradesForTeam(teamIdx, targetTeamIdx) {
    const myTeam   = G.teams[teamIdx];
    const phase    = this.teamPhase(myTeam);
    const proposals = [];

    G.teams.forEach(t => {
      if(t.idx === teamIdx) return;
      if(targetTeamIdx !== undefined && t.idx !== targetTeamIdx) return;

      const theirPhase = this.teamPhase(t);
      // 相互補完性チェック（お互いに欲しいものを持っている）
      if(phase === 'contend' && (theirPhase === 'rebuild' || theirPhase === 'neutral')) {
        // 自分のロールプレイヤー + 指名権 → 相手のスターター
        const myExpend = myTeam.players
          .filter(p => calcOVR(p.abilities) >= 68 && calcOVR(p.abilities) < 78 && p.age >= 27)
          .sort((a,b) => calcOVR(b.abilities) - calcOVR(a.abilities)).slice(0,2);
        const theirTarget = t.players
          .filter(p => calcOVR(p.abilities) >= 76)
          .sort((a,b) => calcOVR(b.abilities) - calcOVR(a.abilities)).slice(0,1);
        if(myExpend.length && theirTarget.length) {
          proposals.push({
            myPlayers: myExpend.map(p=>p.id),
            myPicks: [{round:'1st', fromTeamIdx: teamIdx}],
            theirTeamIdx: t.idx,
            theirPlayers: theirTarget.map(p=>p.id),
            theirPicks: [],
            label: phase + ' → ' + theirPhase,
          });
        }
      }
      if(phase === 'rebuild' && theirPhase === 'contend') {
        // 自分のベテラン → 相手の若手 + 指名権
        const myExpend = myTeam.players
          .filter(p => calcOVR(p.abilities) >= 74 && p.age >= 30)
          .sort((a,b) => calcOVR(b.abilities) - calcOVR(a.abilities)).slice(0,1);
        const theirTarget = t.players
          .filter(p => p.age <= 24 && calcOVR(p.abilities) >= 68)
          .sort((a,b) => calcOVR(b.abilities) - calcOVR(a.abilities)).slice(0,1);
        if(myExpend.length && theirTarget.length) {
          proposals.push({
            myPlayers: myExpend.map(p=>p.id),
            myPicks: [],
            theirTeamIdx: t.idx,
            theirPlayers: theirTarget.map(p=>p.id),
            theirPicks: [{round:'1st', fromTeamIdx: t.idx}],
            label: phase + ' → ' + theirPhase,
          });
        }
      }
    });
    return proposals.slice(0, 5);
  },

  // CPU vs CPU トレード（月次シミュ用）
  runAITradeRound() {
    const teams = G.teams.filter(t => t.idx !== G.myIdx);
    let tradeCount = 0;
    for(const t1 of teams) {
      if(tradeCount >= 3) break;
      const proposals = this.proposeTradesForTeam(t1.idx);
      for(const prop of proposals) {
        const t2 = G.teams[prop.theirTeamIdx];
        if(!t2) continue;
        const t1give  = {players: prop.myPlayers.map(id => t1.players.find(p=>p.id===id)).filter(Boolean), picks: prop.myPicks};
        const t1recv  = {players: prop.theirPlayers.map(id => t2.players.find(p=>p.id===id)).filter(Boolean), picks: prop.theirPicks};
        const accept  = this.shouldAcceptTrade(t2, t1, t1give, t1recv);
        if(!accept.ok) continue;
        // トレード実行
        t1give.players.forEach(p => { t1.players = t1.players.filter(x=>x.id!==p.id); t2.players.push(p); });
        t1recv.players.forEach(p => { t2.players = t2.players.filter(x=>x.id!==p.id); t1.players.push(p); });
        addNews('トレード成立: '+t1.abbr+'/'+t2.abbr,
          (t1give.players[0]?.name||'指名権')+'('+t1.abbr+') → '+(t1recv.players[0]?.name||'指名権')+'('+t2.abbr+')', 'info');
        tradeCount++;
        break;
      }
    }
  },
};

function playerTradeValue(player) {
  return GMBrain.tradeValue(player, null);
}
function pickVal(round, fromIdx) {
  return GMBrain.pickValue(round, fromIdx, 0);
}
function assetTotalValue(teamIdx, playerIds, picks) {
  const team = G.teams[teamIdx];
  const pVal = playerIds.reduce((s,pid) => {
    const p = team.players.find(x=>x.id===pid);
    return s + (p ? GMBrain.tradeValue(p, null) : 0);
  }, 0);
  const kVal = picks.reduce((s,k) => s + GMBrain.pickValue(k.round, k.fromTeamIdx, 0), 0);
  return pVal + kVal;
}
function tradeAIAccept(myTeamIdx, theirTeamIdx, sendVal, recvVal, sendPlayerIds, theirPlayerIds, sendPicks, recvPicks) {
  const myTeam   = G.teams[myTeamIdx];
  const theirTeam= G.teams[theirTeamIdx];
  const iReceive = {
    players: sendPlayerIds.map(id => myTeam.players.find(p=>p.id===id)).filter(Boolean),
    picks:   sendPicks || [],
  };
  const iGive = {
    players: theirPlayerIds.map(id => theirTeam.players.find(p=>p.id===id)).filter(Boolean),
    picks:   recvPicks || [],
  };
  return GMBrain.shouldAcceptTrade(theirTeam, myTeam, iGive, iReceive);
}

function execTrade() {
  if(TS.theirTeamIdx===null) return;
  if(TS.myP.length+TS.myK.length===0||TS.thP.length+TS.thK.length===0){toast('アセットを選択してください','bad');return;}

  const me=G.teams[G.myIdx], them=G.teams[TS.theirTeamIdx];
  const sendSal=trAssetSalary(G.myIdx,TS.myP,TS.myK);
  const recvSal=trAssetSalary(TS.theirTeamIdx,TS.thP,TS.thK);
  const sendVal=assetTotalValue(G.myIdx,TS.myP,TS.myK);
  const recvVal=assetTotalValue(TS.theirTeamIdx,TS.thP,TS.thK);
  const salResult=tradeSalCheck(sendSal,recvSal,TS.useTpe);
  const aiResult=tradeAIAccept(G.myIdx,TS.theirTeamIdx,sendVal,recvVal,TS.myP,TS.thP,TS.myK,TS.thK);
  if(!salResult.ok){toast('Salary: '+salResult.msg,'bad');return;}
  if(!aiResult.ok){toast('AI: '+aiResult.reason,'bad');return;}

  // 選手交換
  const myPlayers=TS.myP.map(pid=>me.players.find(p=>p.id===pid)).filter(Boolean);
  const thPlayers=TS.thP.map(pid=>them.players.find(p=>p.id===pid)).filter(Boolean);
  me.players=me.players.filter(p=>!TS.myP.includes(p.id));
  them.players=them.players.filter(p=>!TS.thP.includes(p.id));
  thPlayers.forEach(p=>me.players.push(p));
  myPlayers.forEach(p=>them.players.push(p));

  // 指名権: G.picks に記録（将来のドラフトで使う）
  if(!G.picks) G.picks={};
  TS.myK.forEach(k=>{ if(!G.picks[them.idx]) G.picks[them.idx]=[]; G.picks[them.idx].push(k); });
  TS.thK.forEach(k=>{ if(!G.picks[G.myIdx]) G.picks[G.myIdx]=[]; G.picks[G.myIdx].push(k); });

  // TPE消費
  if(TS.useTpe&&me.tpe&&recvSal>sendSal) me.tpe=null;

  // TPE生成: 送り出しが受け取りより多い場合、差額がTPEに
  const tpeSal=sendSal-recvSal;
  if(tpeSal>1_000_000) {
    them.tpe=null; // 相手チームはTPEを受け取らない（簡略化）
    // 自チームはTPEを受け取る（大きい選手を送った場合）
    // 実際は受け取る側にTPEが発生するので逆
    me.tpe = {salary:tpeSal, expiresAfter: G.season+1};
    addLog('TPE取得: '+fmtM(tpeSal)+' ('+G.teams[G.myIdx].abbr+')','good');
  }

  const logParts=[];
  if(myPlayers.length) logParts.push(myPlayers.map(p=>p.name).join(',')+' → '+them.abbr);
  if(TS.myK.length) logParts.push(TS.myK.map(k=>k.round+' Pick').join(',')+' → '+them.abbr);
  if(thPlayers.length) logParts.push(thPlayers.map(p=>p.name).join(',')+' → '+me.abbr);
  if(TS.thK.length) logParts.push(TS.thK.map(k=>k.round+' Pick').join(',')+' → '+me.abbr);
  addLog('⇌ Trade: '+logParts.join(' | '),'hl');

  simulateTeamStats(me);
  const got=thPlayers.map(p=>p.name).join(', ')||TS.thK.map(k=>k.round+' Pick').join(', ');
  toast('トレード完了！ 獲得: '+got,'good');
  TS={theirTeamIdx:null,myP:[],myK:[],thP:[],thK:[],useTpe:false};
  renderTrade(); updateHeader();
}

      tabBar.querySelectorAll('button').forEach(b => { b.style.borderBottomColor='transparent'; b.style.color='var(--tx2)'; });

    if(banner && data.G) {

  G.draftOrder=null; G.lotteryDone=false;
  G.taxHistory={}; G.apronHistory={};

}

    const topPlayers = [...t.players].sort((a,b)=>calcOVR(b.abilities)-calcOVR(a.abilities)).slice(0,3);
    const diff = ovr>=78?'Easy':ovr>=73?'Normal':ovr>=68?'Hard':'Very Hard';
    const diffColor = {Easy:'var(--green)',Normal:'var(--blue)',Hard:'var(--gold)','Very Hard':'var(--red)'}[diff];
    const tags=[];
    const avgAge = Math.round(t.players.reduce((s,p)=>s+p.age,0)/t.players.length);
    if(avgAge<=24) tags.push({l:'Young',c:'var(--blue)'});
    if(avgAge>=30) tags.push({l:'Veteran',c:'var(--gold)'});
    const highPot = t.players.filter(p=>p.potential>=1.2).length;
    if(highPot>=3) tags.push({l:'Prospect',c:'var(--green)'});
    const stars = t.players.filter(p=>calcOVR(p.abilities)>=82).length;
    if(stars>=2) tags.push({l:'Star Power',c:'var(--gold)'});
    const payroll = t.payroll;
    const cap = CAP_NEW||140_588_000;
    if(payroll>cap*1.15) tags.push({l:'Over Cap',c:'var(--red)'});
    if(payroll<cap*0.75) tags.push({l:'Cap Room',c:'var(--green)'});
    return {def, idx, t, ovr, topPlayers, diff, diffColor, tags, avgAge, payroll};
  });
}

function dayToMonth(day) {
  let cur = 'Oct';
  for(const mo of MONTH_ORDER) {
    if(day < MONTH_STARTS[mo] + DAYS_IN_MONTH[mo]) { cur = mo; break; }
  }
  return cur;
}
function formatDate(day, season) {
  const mo = dayToMonth(day);
  const d  = day - MONTH_STARTS[mo] + 1;
  const yr = MONTH_STARTS[mo] < MONTH_STARTS['Jan'] ? season : season + 1;
  return MONTH_LABELS[mo] + d + '日 ' + yr;
}

// ── 日程進行 ─────────────────────────────────────────────────
function processDays(fromDay, toDay) {
  // イベント処理
  NBA_EVENTS.filter(e => e.day > fromDay-1 && e.day <= toDay).forEach(e => triggerEvent(e));
  // 試合消化 (day 32-196 = Nov〜Apr)
  const RS=32, RE=196;
  const os=Math.max(fromDay,RS), oe=Math.min(toDay,RE);
  if(oe > os) simulateGamesChunk(Math.round((oe-os) * 82/(RE-RS)));
  G.day = toDay;
  if(G.day > 365) { G.day -= 365; G.season++; G.tradeLocked=false; G.teams.forEach(t=>{t.wins=0;t.losses=0;}); addLog('-- ' + G.season + ' シーズン開始 --','hl'); }
  // 怪我処理（週次）
  if(Math.random() < 0.3) {
    const me = G.teams[G.myIdx];
    const injNews = processInjuries(me);
    injNews.forEach(n => addNews(n.title, n.body, n.type));
  }
  // トレード要求チェック（月次相当）
  if(Math.random() < 0.12) checkTradeRequests();
  // 自動セーブ
  autoSave();
}

function simulateGamesChunk(games) {
  G.teams.forEach(t => {
    const avg = t.players.length ? t.players.reduce((s,p)=>s+calcOVR(p.abilities),0)/t.players.length : 60;
    const str = avg + (t.system.defenseAggression - 0.55)*4;
    const wpct= clamp(0.15+(str-65)*0.026,0.15,0.82);
    const w   = clamp(Math.round(games*wpct + rand(-Math.sqrt(games),Math.sqrt(games))),0,games);
    t.wins   = (t.wins  ||0) + w;
    t.losses = (t.losses||0) + (games-w);
  });
  G.teams.forEach(t => simulateTeamStats(t));
}

// ── プレーオフ演出強化 ──────────────────────────────────────
function simulatePlayoffs(east, west) {
  // 1回戦〜ファイナルをシミュレート
  function simSeries(t1, t2) {
    const s1 = teamAvgOVR(t1) + rand(-3, 3);
    const s2 = teamAvgOVR(t2) + rand(-3, 3);
    return s1 > s2 ? t1 : t2;
  }
  function simBracket(seeds) {
    let bracket = [...seeds];
    const rounds = ['1回戦', '2回戦（CF準決勝）', 'カンファレンス決勝'];
    const results = [];
    while(bracket.length > 1) {
      const next = [];
      const round = rounds[Math.log2(seeds.length/bracket.length)]||'';
      for(let i = 0; i < bracket.length; i+=2) {
        const winner = simSeries(bracket[i], bracket[i+1]);
        next.push(winner);
        results.push({round, winner:winner.name, loser:(winner===bracket[i]?bracket[i+1]:bracket[i]).name});
      }
      bracket = next;
    }
    return {champion:bracket[0], results};
  }

  const eastBracket = simBracket(east.slice(0,8));
  const westBracket = simBracket(west.slice(0,8));
  const champion    = simSeries(eastBracket.champion, westBracket.champion);

  return {champion, eastChamp:eastBracket.champion, westChamp:westBracket.champion,
    eastResults:eastBracket.results, westResults:westBracket.results};
}
const LOTTERY_COMBOS = [140,140,140,125,105,90,75,60,45,30,20,15,10,5];

function runLotteryNBA() {
  const byWinPct = [...G.teams].sort((a,b)=>{
    const pa=a.wins/(a.wins+a.losses||1), pb=b.wins/(b.wins+b.losses||1);
    return pa-pb;
  });
  // 下位14チームがロッタリー対象
  const lottery  = byWinPct.slice(0,14);
  const playoff  = byWinPct.slice(14).reverse(); // 成績良い順→指名順は逆

  // 1000コンビネーションプール
  const pool=[];
  lottery.forEach((t,seed)=>{ const c=LOTTERY_COMBOS[seed]||5; for(let i=0;i<c;i++) pool.push(t.idx); });

  // 上位4位を抽選
  const top=[]; const drawn=new Set();
  while(top.length<Math.min(4,lottery.length)) {
    const pick=pool[Math.floor(Math.random()*pool.length)];
    if(!drawn.has(pick)){drawn.add(pick);top.push(pick);}
  }
  const remaining=lottery.filter(t=>!drawn.has(t.idx)).map(t=>t.idx);
  const playoffPicks=playoff.map(t=>t.idx);
  G.draftOrder=[...top,...remaining,...playoffPicks];
  G.lotteryTeams=lottery.map(t=>t.idx);

  top.forEach((idx,i)=>{
    const t=G.teams[idx];
    const seed=lottery.findIndex(lt=>lt.idx===idx)+1;
    addLog('#'+(i+1)+' 指名権: '+t.abbr+' (Seed'+seed+' / '+(LOTTERY_COMBOS[seed-1]/10).toFixed(1)+'%)',idx===G.myIdx?'good':'');
  });
}

function runAdvance(days) {
  const btn=document.getElementById('btn-adv');
  btn.disabled=true; btn.textContent='処理中...';
  const end=G.day+days;
  let cur=G.day;
  function step(){
    if(cur>=end){updateHeader();renderAll();btn.disabled=false;updateAdvBtn();return;}
    const chunk=Math.min(7,end-cur);
    processDays(cur+1,cur+chunk);
    cur+=chunk;
    setTimeout(step,15);
  }
  setTimeout(step,80);
}

// ── 契約延長ロジック ──────────────────────────────────────────
function canExtend(player) {
  const mo=dayToMonth(G.day);
  if(mo==='Jun'||mo==='Sep') return {ok:false,reason:'オフシーズン交渉期間外'};
  if(player.contract.years<=0) return {ok:false,reason:'契約満了 — FA市場で再契約'};
  if(['Jul','Aug','Oct'].includes(mo)) return {ok:true,reason:''};
  if(player.contract.years===1) return {ok:true,reason:'最終年 — シーズン中も延長可'};
  return {ok:false,reason:'残'+player.contract.years+'年 — プレシーズンのみ延長可'};
}

function calcExtOffer(player) {
  const ovr=calcOVR(player.abilities),age=player.age,cur=player.contract.salary;
  let base;
  if(ovr>=90) base=CAP_NEW*0.35; else if(ovr>=86) base=CAP_NEW*0.28;
  else if(ovr>=82) base=CAP_NEW*0.20; else if(ovr>=78) base=CAP_NEW*0.13;
  else if(ovr>=74) base=CAP_NEW*0.08; else if(ovr>=70) base=CAP_NEW*0.05;
  else base=CAP_NEW*0.025;
  if(age<=23) base*=1.3; else if(age<=25) base*=1.15;
  else if(age>=33) base*=0.75; else if(age>=31) base*=0.88;
  const minSal=Math.max(cur*0.95,MIN_SALARY*2);
  const maxSal=Math.round(Math.min(base,CAP_NEW*0.35));
  const years=ovr>=88&&age<=30?4:ovr>=82?3:age>=32?1:2;
  return {salary:Math.round(clamp((minSal+maxSal)/2,minSal,maxSal)), years, minSal:Math.round(minSal), maxSal};
}

// ── スーパーMAX資格チェック ───────────────────────────────────
// NBA: 直近シーズンにAll-NBA / MVP / DPOY / AllStarMVP受賞で資格
function hasSuperMaxEligibility(player) {
  return !!(player._allNBA || player._mvp || player._dpoy);
}

function getSuperMaxSalary(player) {
  // スーパーMAX: CAP×35%（8年目未満）or 40%（10年目以上）
  const yrs = player._proYears || (player.age - 18);
  if(yrs >= 10) return Math.round((CAP_NEW||140_588_000) * 0.40);
  if(yrs >= 7)  return Math.round((CAP_NEW||140_588_000) * 0.35);
  return Math.round((CAP_NEW||140_588_000) * 0.30);
}

let aiChatHistory = [];

function toggleAIChat() {
  const panel = document.getElementById('ai-chat-panel');
  const fab   = document.getElementById('ai-chat-fab');
  const isOpen = panel.style.display === 'flex';
  if(isOpen) {
    panel.style.display = 'none';
  } else {
    panel.style.display = 'flex';
    // FABの位置に合わせてパネルを表示
    const fabLeft = parseInt(fab.style.left) || (window.innerWidth - 68);
    const fabTop  = parseInt(fab.style.top)  || (window.innerHeight - 160);
    const panelW  = Math.min(window.innerWidth - 24, 380);
    // FABの左か右にパネルを出す
    let panelLeft = fabLeft + 60; // 右側
    if(panelLeft + panelW > window.innerWidth - 8) panelLeft = fabLeft - panelW - 8; // 左側
    panelLeft = clamp(panelLeft, 8, window.innerWidth - panelW - 8);
    // FABより上にパネルを出す
    const panelMaxH = 380;
    let panelTop = fabTop - panelMaxH - 8;
    if(panelTop < 60) panelTop = fabTop + 60;
    panelTop = clamp(panelTop, 60, window.innerHeight - panelMaxH - 8);
    panel.style.left   = panelLeft + 'px';
    panel.style.top    = panelTop + 'px';
    panel.style.width  = panelW + 'px';
    panel.style.maxHeight = panelMaxH + 'px';
    setTimeout(()=>{ document.getElementById('ai-chat-input')?.focus(); }, 100);
  }
}

async function sendAIChat() {
  const input = document.getElementById('ai-chat-input');
  const messages = document.getElementById('ai-chat-messages');
  const text = (input?.value||'').trim();
  if(!text) return;

  // ユーザーメッセージ
  const userMsg = document.createElement('div');
  userMsg.className = 'ai-msg user';
  userMsg.textContent = text;
  messages.appendChild(userMsg);
  input.value = '';

  // タイピング表示
  const typingMsg = document.createElement('div');
  typingMsg.className = 'ai-msg ai typing';
  typingMsg.id = 'ai-typing';
  typingMsg.textContent = '考え中...';
  messages.appendChild(typingMsg);
  messages.scrollTop = messages.scrollHeight;

  aiChatHistory.push({role:'user', content:text});
  if(aiChatHistory.length > 20) aiChatHistory.splice(0,2);

  // APIコール
  try {
    const resp = await fetch('https://api.anthropic.com/v1/messages', {
      method:'POST',
      headers:{
        'Content-Type':'application/json',
        'x-api-key': '',
        'anthropic-version': '2023-06-01',
        'anthropic-dangerous-direct-browser-access': 'true',
      },
      body: JSON.stringify({
        model:'claude-sonnet-4-20250514',
        max_tokens:400,
        system: AI_SYSTEM_PROMPT,
        messages: aiChatHistory,
      })
    });
    const data = await resp.json();
    const reply = data.content?.[0]?.text || '回答を取得できませんでした。';
    aiChatHistory.push({role:'assistant', content:reply});

    const el = document.getElementById('ai-typing');
    if(el) { el.className='ai-msg ai'; el.textContent=reply; el.id=''; }
  } catch(e) {
    const el = document.getElementById('ai-typing');
    if(el) { el.textContent='接続エラー: '+e.message; el.id=''; }
  }
  messages.scrollTop = messages.scrollHeight;
}

// FABドラッグ機能
function initFabDrag() {
  const fab = document.getElementById('ai-chat-fab');
  if(!fab) return;

  let isDragging = false;
  let startX, startY, fabX, fabY;
  // 初期位置（右下）
  let posX = window.innerWidth  - 68;
  let posY = window.innerHeight - 160;
  fab.style.right = 'auto';
  fab.style.bottom = 'auto';
  fab.style.left = posX + 'px';
  fab.style.top  = posY + 'px';

  fab.addEventListener('touchstart', e => {
    const t = e.touches[0];
    startX = t.clientX; startY = t.clientY;
    fabX = parseInt(fab.style.left)||posX;
    fabY = parseInt(fab.style.top) ||posY;
    isDragging = false;
  }, {passive:true});

  fab.addEventListener('touchmove', e => {
    const t = e.touches[0];
    const dx = t.clientX - startX;
    const dy = t.clientY - startY;
    if(Math.abs(dx)>5 || Math.abs(dy)>5) isDragging = true;
    if(!isDragging) return;
    e.preventDefault();
    const nx = clamp(fabX + dx, 8, window.innerWidth  - 60);
    const ny = clamp(fabY + dy, 8, window.innerHeight - 60);
    fab.style.left = nx + 'px';
    fab.style.top  = ny + 'px';
    posX = nx; posY = ny;
  }, {passive:false});

  fab.addEventListener('touchend', e => {
    if(!isDragging) toggleAIChat(); // ドラッグなし→タップ
  });

  // パネルの位置もFABに追従
  const panel = document.getElementById('ai-chat-panel');
  if(panel) {
    panel.style.right = 'auto';
    panel.style.bottom = 'auto';
  }
}

// チャットイベント登録（startGame後に実行）
function initAIChatListeners() {
  const fab   = document.getElementById('ai-chat-fab');
  const close = document.getElementById('ai-chat-close');
  const send  = document.getElementById('ai-chat-send');
  const input = document.getElementById('ai-chat-input');
  if(close) close.addEventListener('click', toggleAIChat);
  if(send)  send.addEventListener('click', sendAIChat);
  if(input) input.addEventListener('keydown', e=>{ if(e.key==='Enter') sendAIChat(); });
  initFabDrag();
}

let tutorialStep = 0;

function showTutorial() {
  const overlay = document.getElementById('tutorial-overlay');
  const body    = document.getElementById('tutorial-body');
  if(!overlay || !body) return;

  tutorialStep = -1; // 選択画面
  body.innerHTML = '';

  // 選択画面
  const sel = document.createElement('div');
  sel.style.cssText = 'padding-top:8px';
  sel.innerHTML =
    '<div style="text-align:center;padding:20px 0 16px">'+
      '<div style="font-size:48px;margin-bottom:12px">🏀</div>'+
      '<div style="font-family:\'Barlow Condensed\',sans-serif;font-size:28px;font-weight:900;color:var(--gold);margin-bottom:8px">GM Sim</div>'+
      '<div style="font-size:15px;color:var(--tx2);line-height:1.6">NBAチームのGMとして<br>チャンピオンを目指そう</div>'+
    '</div>';

  const optWrap = document.createElement('div');
  optWrap.style.cssText = 'display:flex;flex-direction:column;gap:10px;margin-bottom:16px';

  const opts = [
    {key:'full',  label:'フルチュートリアル',    sub:'全8ステップ（約3分）',          color:'var(--gold)',  bg:'rgba(232,160,32,.1)', border:'rgba(232,160,32,.3)'},
    {key:'quick', label:'クイックスタート',       sub:'重要ポイントのみ（約1分）',      color:'var(--blue)',  bg:'rgba(74,158,255,.08)', border:'rgba(74,158,255,.2)'},
    {key:'skip',  label:'スキップしてプレイ',     sub:'すぐに始める',                   color:'var(--tx2)',   bg:'transparent',          border:'var(--bdr2)'},
  ];

  opts.forEach(opt => {
    const btn = document.createElement('button');
    btn.style.cssText = 'width:100%;background:'+opt.bg+';border:1px solid '+opt.border+';border-radius:12px;padding:16px;text-align:left;cursor:pointer;-webkit-appearance:none';
    btn.innerHTML =
      '<div style="font-family:\'Barlow Condensed\',sans-serif;font-size:18px;font-weight:700;color:'+opt.color+'">'+opt.label+'</div>'+
      '<div style="font-size:12px;color:var(--tx3);margin-top:3px">'+opt.sub+'</div>';
    btn.addEventListener('click', () => {
      tutorialMode = opt.key;
      if(opt.key === 'skip') { closeTutorial(); return; }
      renderTutorialStep(0);
    });
    optWrap.appendChild(btn);
  });

  sel.appendChild(optWrap);
  body.appendChild(sel);
  overlay.classList.add('open');
}

function renderTutorialStep(idx) {
  const overlay = document.getElementById('tutorial-overlay');
  const body    = document.getElementById('tutorial-body');

  // クイックモードは重要ステップのみ（0,2,3,4,7）
  const quickSteps = [0,2,3,4,7];
  const steps = tutorialMode==='quick'
    ? TUTORIAL_STEPS.filter((_,i)=>quickSteps.includes(i))
    : TUTORIAL_STEPS;

  if(idx >= steps.length) { closeTutorial(); return; }
  tutorialStep = idx;
  const step = steps[idx];
  body.innerHTML = '';
  overlay.classList.add('open');

  // プログレスバー
  const prog = document.createElement('div');
  prog.style.cssText = 'display:flex;gap:4px;padding:8px 0 14px';
  steps.forEach((_,i) => {
    const dot = document.createElement('div');
    dot.style.cssText = 'height:3px;flex:1;border-radius:2px;background:'+(i<=idx?'var(--gold)':'var(--bdr2)');
    prog.appendChild(dot);
  });
  body.appendChild(prog);

  // コンテンツ
  const content = document.createElement('div');
  content.innerHTML =
    '<div style="font-size:40px;text-align:center;margin-bottom:14px">'+step.icon+'</div>'+
    '<div style="font-family:\'Barlow Condensed\',sans-serif;font-size:24px;font-weight:900;text-align:center;margin-bottom:14px;color:var(--gold)">'+step.title+'</div>'+
    '<div style="font-size:15px;color:var(--tx);line-height:1.75;white-space:pre-line;margin-bottom:20px">'+step.content+'</div>';
  body.appendChild(content);

  // ボタン
  const btnRow = document.createElement('div');
  btnRow.style.cssText = 'display:flex;gap:10px';

  if(idx > 0) {
    const back = document.createElement('button');
    back.style.cssText = 'flex:1;background:var(--surf2);color:var(--tx2);border:1px solid var(--bdr);border-radius:8px;padding:14px;font-family:\'Barlow Condensed\',sans-serif;font-size:15px;font-weight:700;cursor:pointer';
    back.textContent = '戻る';
    back.addEventListener('click', () => renderTutorialStep(idx-1));
    btnRow.appendChild(back);
  }

  const next = document.createElement('button');
  const isLast = idx === steps.length-1;
  next.style.cssText = 'flex:2;background:var(--gold);color:#000;border:none;border-radius:8px;padding:14px;font-family:\'Barlow Condensed\',sans-serif;font-size:16px;font-weight:700;cursor:pointer';
  next.textContent = isLast ? 'ゲームを始める！' : '次へ ('+(idx+1)+'/'+steps.length+')';
  next.addEventListener('click', () => renderTutorialStep(idx+1));
  btnRow.appendChild(next);

  // スキップ（最後以外）
  if(!isLast) {
    const skip = document.createElement('button');
    skip.style.cssText = 'background:none;border:none;color:var(--tx3);font-size:13px;cursor:pointer;padding:8px;margin-top:6px;width:100%;text-align:center';
    skip.textContent = 'スキップして始める';
    skip.addEventListener('click', closeTutorial);
    body.appendChild(btnRow);
    body.appendChild(skip);
    return;
  }

  body.appendChild(btnRow);
}

function closeTutorial() {
  document.getElementById('tutorial-overlay')?.classList.remove('open');
  localStorage.setItem('gmSimTutorialDone','1');
}

// startGame後のUI初期化
function showGameUI() {
  document.getElementById('ai-chat-fab').style.display = 'flex';
  initAIChatListeners();
  // チーム選択前にtutorialMode='full'か'quick'が設定されていればそのまま続ける
  if(tutorialMode === 'full' || tutorialMode === 'quick') {
    setTimeout(()=>renderTutorialStep(0), 400);
  } else if(!localStorage.getItem('gmSimTutorialDone')) {
    // 初回起動でチュートリアルを選んでいない場合は選択画面を出す
    setTimeout(showTutorial, 400);
  }
}
